model { 
 for(i in 1 : N)
      {
        poids[i] ~ dnorm(mu[i], tau)
        mu[i] <- moyennes[sexe[i]] + b * (nbsemaines[i]-nbsemaines.bar) + c * (prisepoids[i]-prisepoids.bar)
      }
      b ~ dunif(0,500)
      nbsemaines.bar <- mean(nbsemaines)
      c ~ dunif(-500,500)
      prisepoids.bar <- mean(prisepoids)
      moyennes[1] ~ dunif(2500, 5000)
      moyennes[2] ~ dunif(2500, 5000)
      tau <- 1/(sigma^2)
      sigma ~ dunif(200, 800)
}
